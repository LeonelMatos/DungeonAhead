163 private void UseItem(Item item) //Use Items utilities here!
164 {
165 windowTooltip.HideTooltip_Public();
166 switch (item.itemType)
167 {
168 case Item.ItemType.HealthPotion:
169 //Debug.Log("Used HealthPotion");
170 if(playerStats.health != playerStats.maxHealth)
171 {
172 RemoveUsedItem(Item.ItemType.HealthPotion);
173 playerStats.TakeDamage(-20);
174 }
175 else
176 StartCoroutine(maxValueAnimation(GameObject.FindGameObjectWithTag("HealthBar/Icon"))); //Full bar anim.
177 break;
178 case Item.ItemType.EnergyPotion:
179 //Debug.Log("Used EnergyPotion");
180 if(playerStats.energy != playerStats.maxEnergy)
181 {
182 RemoveUsedItem(Item.ItemType.EnergyPotion);
183 playerStats.TakeEnergy(-20);
184 }
185 else
186 StartCoroutine(maxValueAnimation(GameObject.FindGameObjectWithTag("EnergyBar/Icon")));
187 break;
188 case Item.ItemType.Medkit:
189 //Debug.Log("Used Medkit");
190 if(playerStats.health < playerStats.maxHealth)
191 {
192 RemoveUsedItem(Item.ItemType.Medkit);
193 RemoveUsedItem2(item);
194 playerStats.TakeDamage(-50);
195 }
196 else
197 {
198 StartCoroutine(maxValueAnimation(GameObject.FindGameObjectWithTag("HealthBar/Icon")));
199 }
200 break;
201 case Item.ItemType.NightVisionPotion:
202 //Debug.Log("Used NightVisionPotion");
203 RemoveUsedItem(Item.ItemType.NightVisionPotion);
204 effects.SetEffect(1); //Default: 180 (3 min.)
205 break;
206 case Item.ItemType.Milk:
207 //Debug.Log("Used Milk");
208 RemoveUsedItem(Item.ItemType.Milk);
209 effects.UseMilk();
210 break;
211 case Item.ItemType.SpeedPotion:
212 //Debug.Log("Used SpeedPotion");
213 RemoveUsedItem(Item.ItemType.SpeedPotion);
214 effects.SetEffect(2); //Default: 180 (3 min.)
215 break;
216 case Item.ItemType.Book: //BOOK
217 textWindow.GetComponent<TextWindow>().OpenTextWindow(item, inventoryUI);
218 break;
219 case Item.ItemType.RandomPotion:
220 effects.UseRandomPotion();
221 RemoveUsedItem(Item.ItemType.RandomPotion);
222 break;
223 }
224 }
